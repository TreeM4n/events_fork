trait_mem_ancestors_grudge_archaeotech_excellence = {
	cost = 4
	allowed_archetypes = {
		ROBOT
		MACHINE
	}
	species_potential_add = {
		from = {
			has_technology = tech_mem_ancestors_grudge_robot_traits
		}
		#can_do_mem_specialized_robo_modding = yes
	}
	species_possible_remove = {
		always = yes
	}
	species_possible_merge_remove = {
		always = yes
	}
	
	initial = no
	randomized = no
	modifier = {
		planet_jobs_produces_mult = 0.2
		planet_jobs_upkeep_mult = 0.3
		#planet_pop_assembly_mult = -0.3
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				has_mining_designation = yes
				has_rural_designation = yes
			}
		}
		planet_miners_minerals_produces_add = 1
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				has_farming_designation = yes
				has_rural_designation = yes
			}
		}
		planet_farmers_food_produces_add = 1
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				has_generator_designation = yes
				has_rural_designation = yes
			}
		}
		planet_technician_energy_produces_add = 1
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				OR = {
					has_foundry_designation = yes
					has_industrial_designation = yes
				}
			}
		}
		planet_metallurgists_alloys_produces_add = 1
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				OR = {
					has_factory_designation = yes
					has_industrial_designation = yes
				}
			}
		}
		planet_artisans_consumer_goods_produces_add = 1
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				has_research_designation = yes
			}
		}
		planet_researchers_physics_research_produces_add = 0.5
		planet_researchers_society_research_produces_add = 0.5
		planet_researchers_engineering_research_produces_add = 0.5
	}
	triggered_pop_modifier = {
		potential = {
			planet = {
				has_unity_designation = yes
			}
		}
		planet_administrators_unity_produces_add = 1
	}
	slave_cost = {
		energy = 1250
	}
	assembly_score = {
		base = 4
	}
	custom_tooltip_with_modifiers = automodding_trait_mem_robotic_tooltip
}
